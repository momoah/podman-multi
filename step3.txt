Step 3 - Add notes-proxy (nginx) to the notes pod
=================================================

# in the notes/ directory:

cat > nginx.conf <<'NGINX'
# Minimal, production-leaning defaults for an in-pod reverse proxy
server {
    # Unprivileged nginx image listens on 8080 by default
    listen 8080;

    # Static root not used here, but harmless if present
    root /usr/share/nginx/html;

    # Health endpoint (no upstream call needed, but weâ€™ll pass-through for API parity)
    location = /healthz {
        proxy_pass http://127.0.0.1:8000/healthz;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header Connection "";
    }

    # Everything else to the API
    location / {
        proxy_pass http://127.0.0.1:8000/;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Forwarded-For $remote_addr;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Connection "";
        client_max_body_size 10m;
        proxy_read_timeout 60s;
        proxy_send_timeout 60s;
    }
}
NGINX
